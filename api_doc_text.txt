    document.getElementById('btnProceedviaRazorpay').onclick = function (e) {
        $.ajax({
            type: "Post",

            headers: {
                "Authorization": "Bearer <JWT Token>"

d
ata: 
“place json body”      
      
Success: function (r) {
                var orderId = r.Msg;
                var shoppingCartId = r.Status;
                var options = {
                    "key": "place test key",
                    "name": "AstroVed",
                    "order_id": orderId,
                    "image": "https://www.astroved.com/Images/astroved-logo.jpg",
                    "prefill": {
                        "contact": "place mobile number",
                        "email": "place emailId",
                    },
                    "theme": {
                        "color": "#4e89df"
                    },
                    // "method": {
                    //     "card": false,
                    //     "netbanking": false,
                    //     "wallet": false,
                    // },
                    "readonly":{
                        "name":true,
                        "contact":true,
                        "email": true,
                    }
                }
                options.theme.image_padding = false;
                options.handler = function (response) {
                    $("#btnProceedviaRazorpay").css("display", "none");
                    document.getElementById('razorpay_payment_id').value = response.razorpay_payment_id;
                    document.getElementById('razorpay_order_id').value = orderId;
                    document.getElementById('razorpay_signature').value = response.razorpay_signature;
                    var notes = {
                        OrderId: shoppingCartId
                    };
                    document.getElementById('razorpay_notes').value = JSON.stringify(notes);
                    document.razorpayForm.action = 'place URL after complete payment';
                    document.razorpayForm.submit();
                };
                options.modal = {
                    ondismiss: function () {
                        console.log("This code runs when the popup is closed");
                    },
                    escape: true,
                    backdropclose: false
                };
                var rzp = new Razorpay(options);
                rzp.open();
                e.preventDefault();
            },
            failure: function (r) {
                $('#RazorpayPolicy').show();
                $('#RazorpayPolicy').empty();
                $('#RazorpayPolicy').append(r.d);
            }
        });
    }
